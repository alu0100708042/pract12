    class Matrix

       # Constructor de la clase Matrix. Recibe row y col como parametros para asignar las filas y columnas de la matriz.
       def initialize(row, col)
               @row = row
               @col = col
       end

       # Asignación del método get para :row y :col
       attr_accessor :row, :col

    end
